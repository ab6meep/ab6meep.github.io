    const text = 'Angel';
    const typeEl = document.querySelector('.typewriter');
    let index = 0;
    let forward = true;

    function updateText() {
      if (forward) {
        index += 1;
        typeEl.textContent = text.slice(0, index);
        if (index === text.length) {
          forward = false;
          setTimeout(updateText, 2000);
          return;
        }
      } else {
        index -= 1;
        typeEl.textContent = text.slice(0, index);
        if (index === 1) {
          forward = true;
          setTimeout(updateText, 800);
          return;
        }
      }
      setTimeout(updateText, forward ? 160 : 110);
    }

    updateText();

    // Click to enter
    const enterScreen = document.getElementById('enter-screen');
    enterScreen.addEventListener('click', () => {
      enterScreen.classList.add('fade-out');
      audio.play().then(() => {
        playBtn.textContent = '⏸ Pause';
      }).catch(() => {
        // autoplay may still be blocked by browser
      });
      setTimeout(() => {
        enterScreen.classList.add('hidden');
      }, 500);
    });

    // Audio player controls
    const audio = document.getElementById('audio-player');
    const playBtn = document.getElementById('play-pause-btn');
    const progressBar = document.getElementById('progress-bar');
    const timeCurrent = document.getElementById('time-current');
    const timeTotal = document.getElementById('time-total');

    function formatTime(seconds) {
      if (!seconds || isNaN(seconds)) return '00:00';
      const mins = Math.floor(seconds / 60);
      const secs = Math.floor(seconds % 60);
      return `${String(mins).padStart(2, '0')}:${String(secs).padStart(2, '0')}`;
    }

    // Initialize progress bar at 0
    progressBar.value = 0;
    progressBar.style.setProperty('--progress', '0%');
    timeCurrent.textContent = '00:00';
    timeTotal.textContent = '00:00';

    // Reset audio to start
    audio.currentTime = 0;
    audio.volume = 0.18;

    // Try to start autoplay
    audio.play().catch(() => {
      // If autoplay fails (browser policy), just set up the button
    });

    playBtn.addEventListener('click', () => {
      if (audio.paused) {
        audio.play();
        playBtn.textContent = '⏸ Pause';
      } else {
        audio.pause();
        playBtn.textContent = '▶ Play';
      }
    });

    audio.addEventListener('play', () => {
      playBtn.textContent = '⏸ Pause';
    });

    audio.addEventListener('pause', () => {
      playBtn.textContent = '▶ Play';
    });

    audio.addEventListener('timeupdate', () => {
      if (audio.duration) {
        const percent = (audio.currentTime / audio.duration) * 100;
        progressBar.value = percent;
        progressBar.style.setProperty('--progress', percent + '%');
        timeCurrent.textContent = formatTime(audio.currentTime);
      }
    });

    audio.addEventListener('loadedmetadata', () => {
      timeTotal.textContent = formatTime(audio.duration);
    });

    function createRainDrops(container, count) {
      for (let i = 0; i < count; i += 1) {
        const drop = document.createElement('div');
        const sizeClass = Math.random() > 0.65 ? 'long' : '';
        const speedClass = Math.random() > 0.5 ? 'fast' : 'slow';
        drop.className = `drop ${sizeClass} ${speedClass}`.trim();
        drop.style.left = `${Math.random() * 100}%`;
        drop.style.animationDuration = `${0.7 + Math.random() * 0.9}s`;
        drop.style.animationDelay = `${-Math.random() * 2}s`;
        container.appendChild(drop);
      }
    }

        progressBar.style.setProperty('--progress', progressBar.value + '%');
      }
    });
